

services:

  mariadb:
    image: mariadb:10.6
    container_name: mariadb
    hostname: mariadb
    restart: always
    environment:
      MARIADB_ROOT_PASSWORD: root
      MARIADB_DATABASE: metastore_db
      MARIADB_USER: hive
      MARIADB_PASSWORD: hive
    healthcheck:
      test: [ "CMD-SHELL", "mariadb-admin ping -h 127.0.0.1 -uroot -proot || exit 1" ]
      interval: 10s
      timeout: 5s
      retries: 15
      start_period: 20s
    ports:
      - "3306:3306"
    volumes:
      - ./mariadb_data:/var/lib/mysql
    networks:
      - lakehouse

  metastore:
    image: apache/hive:4.0.0
    container_name: hive-metastore
    hostname: metastore
    depends_on:
      mariadb:
        condition: service_healthy
    environment:
      SERVICE_NAME: metastore
      DB_DRIVER: mysql
    ports:
      - "9083:9083"
    volumes:
      - ./warehouse:/opt/hive/data/warehouse
      - ./mysql-connector-j-8.0.33.jar:/opt/hive/lib/mysql-connector-j.jar
      - ./hive/conf/core-site.xml:/opt/hive/conf/core-site.xml
      - ./hive/conf/metastore-site.xml:/opt/hive/conf/metastore-site.xml
    networks:
      - lakehouse


  trino:
    image: trinodb/trino:latest
    container_name: trino
    hostname: trino
    depends_on:
      - metastore
    ports:
      - "8080:8080"
    volumes:
      - ./trino/etc:/etc/trino
    networks:
      - lakehouse

volumes:
  mariadb_data:
  warehouse:

networks:
  lakehouse:
